package JavaAIPractice;

import java.util.Scanner;

public class ReverseDisplayLowest {
    
    public static void displayArray(int[] arr) {
        System.out.print("Numbers: ");
        for (int num : arr) {
            System.out.print(num + " ");
        }
        System.out.println();
    }

    public static void reverseArray(int[] arr) {
        int start = 0, end = arr.length - 1;
        while (start < end) {
            int temp = arr[start];
            arr[start] = arr[end];
            arr[end] = temp;
            start++;
            end--;
        }
    }

    // Method to find the lowest number using conditional statement
    public static int findLowest(int[] arr) {
        int lowest = arr[0];
        for (int i = 1; i < arr.length; i++) {
            // Conditional statement to check if a new lowest value is found
            if (arr[i] < lowest) {
                lowest = arr[i];
            }
        }
        return lowest;
    }

    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        System.out.print("Enter how many numbers: ");
        int n = sc.nextInt();
        int[] numbers = new int[n];

        for (int i = 0; i < n; i++) {
            System.out.print("Enter number " + (i + 1) + ": ");
            numbers[i] = sc.nextInt();
        }

        System.out.println("\n--- Original Order ---");
        displayArray(numbers);

        reverseArray(numbers);
        System.out.println("\n--- Reversed Order ---");
        displayArray(numbers);

        int lowest = findLowest(numbers);
        System.out.println("\nLowest Number: " + lowest);
    }
}
