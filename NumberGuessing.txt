package javaai;

import java.util.Random;
import java.util.Scanner;

public class NumberGuessing {

    // Method to generate a random number
    public static int generateNumber() {
        Random rand = new Random();
        return rand.nextInt(100) + 1; // random number between 1 and 100
    }

    // Method to play the game
    public static void playGame(int[] attemptsRecord, Scanner sc) {
        int secret = generateNumber();
        int guess;
        int attempts = 0;

        System.out.println("\nðŸŽ® Welcome to the Number Guessing Game!");
        System.out.println("I have picked a number between 1 and 100. Try to guess it!");

        do {
            System.out.print("Enter your guess: ");
            guess = sc.nextInt();
            attempts++;

            if (guess < secret) {
                System.out.println("Too LOW! Try again.");
            } else if (guess > secret) {
                System.out.println("Too HIGH! Try again.");
            } else {
                System.out.println("ðŸŽ‰ Correct! You guessed it in " + attempts + " attempts.");
                break;
            }

        } while (guess != secret);

        // Save attempts in array (history)
        attemptsRecord[0] = attempts;
    }

    // Method to display attempts history
    public static void displayHistory(int[] attemptsRecord) {
        if (attemptsRecord[0] == 0) {
            System.out.println("No game played yet!");
        } else {
            System.out.println("Your last game took " + attemptsRecord[0] + " attempts.");
        }
    }

    // Main method
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int[] attemptsRecord = {0}; // array to store attempts history
        int choice;

        do {
            // Menu
            System.out.println("\n===== Number Guessing Game Menu =====");
            System.out.println("1. Play Game");
            System.out.println("2. View Last Game Attempts");
            System.out.println("3. Exit");
            System.out.print("Choose an option: ");
            choice = sc.nextInt();

            switch (choice) {
                case 1:
                    playGame(attemptsRecord, sc);
                    break;
                case 2:
                    displayHistory(attemptsRecord);
                    break;
                case 3:
                    System.out.println("Thanks for playing! Goodbye ðŸ‘‹");
                    break;
                default:
                    System.out.println("Invalid choice!");
            }
        } while (choice != 3);

        sc.close();
    }
}
