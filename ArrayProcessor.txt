package javaai;

import java.util.Scanner;

public class ArrayProcessor {

    // Method to input numbers into the array
    public static void inputNumbers(int[] arr, Scanner sc) {
        System.out.println("Enter " + arr.length + " numbers:");
        for (int i = 0; i < arr.length; i++) {
            System.out.print("Number " + (i + 1) + ": ");
            arr[i] = sc.nextInt();
        }
    }

    // Method to find maximum number
    public static int findMax(int[] arr) {
        int max = arr[0];
        for (int num : arr) {
            if (num > max) {
                max = num;
            }
        }
        return max;
    }

    // Method to find minimum number
    public static int findMin(int[] arr) {
        int min = arr[0];
        for (int num : arr) {
            if (num < min) {
                min = num;
            }
        }
        return min;
    }

    // Method to calculate average
    public static double calculateAverage(int[] arr) {
        int sum = 0;
        for (int num : arr) {
            sum += num;
        }
        return (double) sum / arr.length;
    }

    // Method to display even/odd classification
    public static void displayEvenOdd(int[] arr) {
        System.out.println("\nEven and Odd Classification:");
        for (int num : arr) {
            if (num % 2 == 0) {
                System.out.println(num + " is EVEN");
            } else {
                System.out.println(num + " is ODD");
            }
        }
    }

    // Main method
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        // Ask user how many numbers
        System.out.print("How many numbers do you want to enter? ");
        int size = sc.nextInt();

        int[] numbers = new int[size];

        // Input numbers
        inputNumbers(numbers, sc);

        // Display results
        System.out.println("\n===== Results =====");
        System.out.println("Maximum number: " + findMax(numbers));
        System.out.println("Minimum number: " + findMin(numbers));
        System.out.println("Average: " + calculateAverage(numbers));

        // Display even/odd classification
        displayEvenOdd(numbers);

        sc.close();
    }
}
