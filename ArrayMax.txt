package javaai;
import java.util.Arrays;
import java.util.Scanner;

public class ArrayMax {
    
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        
        System.out.print("Enter how many numbers: ");
        int n = sc.nextInt();

        int[] numbers = new int[n];

        System.out.println("Enter " + n + " numbers:");
        for (int i = 0; i < n; i++) {
            numbers[i] = sc.nextInt();
        }

        int max = numbers[0];
        int min = numbers[0];

        for (int i = 1; i < numbers.length; i++) {
            if (numbers[i] > max) max = numbers[i];
            if (numbers[i] < min) min = numbers[i];
        }

        Arrays.sort(numbers);
        double middleValue;
        if (numbers.length % 2 == 0) {
            middleValue = (numbers[numbers.length / 2 - 1] + numbers[numbers.length / 2]) / 2.0;
        } else {
            middleValue = numbers[numbers.length / 2];
        }

        System.out.println("\nThe maximum value in the array is: " + max);
        System.out.println("The minimum value in the array is: " + min);
        System.out.println("The median value in the array is: " + middleValue);

        sc.close();
    }
}
