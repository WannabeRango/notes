package Practice;

import java.util.Scanner;

public class SearchNameLinkedList {

	    // Method to input names
	    public static void inputNames(String[] names, Scanner sc) {
	        System.out.println("\nEnter " + names.length + " names:");
	        for (int i = 0; i < names.length; i++) {
	            System.out.print("Enter name " + (i + 1) + ": ");
	            names[i] = sc.nextLine();
	        }
	    }

	    // Method to display all names
	    public static void displayNames(String[] names) {
	        System.out.println("\n--- Names in the List ---");
	        for (int i = 0; i < names.length; i++) {
	            System.out.println((i + 1) + ". " + names[i]);
	        }
	    }

	    // Method to search a name
	    public static void searchName(String[] names, String search) {
	        boolean found = false;
	        for (int i = 0; i < names.length; i++) {
	            if (names[i].equalsIgnoreCase(search)) {
	                System.out.println("Name \"" + search + "\" found at position " + (i + 1));
	                found = true;
	                break;
	            }
	        }
	        if (!found) {
	            System.out.println("No Name Found");
	        }
	    }

	    // Main method
	    public static void main(String[] args) {
	        Scanner sc = new Scanner(System.in);

	        // Step 1: Ask for size
	        System.out.print("Enter the size of the array: ");
	        int size = sc.nextInt();
	        sc.nextLine(); // consume newline

	        // Step 2: Create array
	        String[] names = new String[size];

	        // Step 3: Input names
	        inputNames(names, sc);

	        // Step 4: Menu loop
	        int choice;
	        do {
	            System.out.println("\n--- MENU ---");
	            System.out.println("[1] View All Names");
	            System.out.println("[2] Search a Name");
	            System.out.println("[0] Exit");
	            System.out.print("Enter your choice: ");
	            choice = sc.nextInt();
	            sc.nextLine(); // consume newline

	            if (choice == 1) {
	                displayNames(names);
	            } else if (choice == 2) {
	                System.out.print("Enter a name to search: ");
	                String search = sc.nextLine();
	                searchName(names, search);
	            } else if (choice == 0) {
	                System.out.println("Exiting program...");
	            } else {
	                System.out.println("Invalid choice. Try again.");
	            }

	        } while (choice != 0);

	        sc.close();
	    }
	}
