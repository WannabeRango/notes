public class MultiplyWithoutAsterisk {
    public static void main(String[] args) {
        int num1 = 9;
        int num2 = 9;
        int result = 0;

        // Repeat addition
        for (int i = 0; i < num2; i++) {
            result += num1;
        }

        System.out.println(num1 + " x " + num2 + " = " + result);
    }
}
